CREATE TABLE [dbo].[Users] (
    [Id]       INT              IDENTITY (1, 1) NOT NULL,
    [Login]    CHAR (17)        NOT NULL,
    [Password] UNIQUEIDENTIFIER NOT NULL,
    [UserType] NVARCHAR (16)    NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    UNIQUE NONCLUSTERED ([Login] ASC)
);


CREATE TABLE [dbo].[Tests] (
    [Id]     INT            IDENTITY (1, 1) NOT NULL,
    [Header] NVARCHAR (226) NOT NULL,
    [Topic]  NVARCHAR (226) NOT NULL,
    [Author] CHAR (17)      NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);



CREATE TABLE [dbo].[Questions] (
    [Id]       INT      IDENTITY (1, 1)      NOT NULL,
    [Question] NVARCHAR (226) NOT NULL,
	[Cost]	   INT	 ,
	[Image]    NVARCHAR(226)  NULL,
    [Test_Id]  INT            NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Questions_Tests] FOREIGN KEY ([Test_Id]) REFERENCES [dbo].[Tests] ([Id])
);


CREATE TABLE [dbo].[Answers] (
    [Id]          INT     IDENTITY (1, 1)  NOT NULL,
    [Answer]      NVARCHAR (226) NOT NULL,
    [IsRight]     BIT            NOT NULL,
    [Question_Id] INT            NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Answers_Questions] FOREIGN KEY ([Question_Id]) REFERENCES [dbo].[Questions] ([Id])
);
